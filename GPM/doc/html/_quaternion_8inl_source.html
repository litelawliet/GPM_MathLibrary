559  inline Vector3<double> Quaternion::ToEuler() const
560  {
561  Vector3<double> euler{};
562 
563  // roll (x-axis rotation)
564  const double sinr_cosp = 2.0 * (w * axis.x + axis.y * axis.z);
565  const double cosr_cosp = 1.0 - 2.0 * (axis.x * axis.x + axis.y * axis.y);
566  euler.x = Tools::Utils::Arctan2(sinr_cosp, cosr_cosp);
567 
568  // pitch (y-axis rotation)
569  const double sinp = 2.0 * (w * axis.y - axis.z * axis.x);
570  if (Tools::Utils::Abs(sinp) >= 1.0)
571  euler.y = std::copysign(Tools::M_PI / 2.0, sinp); // use 90 degrees if out of range
572  else
573  euler.y = Tools::Utils::Arcsin(sinp);
574 
575  // yaw (z-axis rotation)
576  const double siny_cosp = 2.0 * (w * axis.z + axis.x * axis.y);
577  const double cosy_cosp = 1.0 - 2.0 * (axis.y * axis.y + axis.z * axis.z);
578  euler.z = Tools::Utils::Arctan2(siny_cosp, cosy_cosp);
579 
580  euler.x = Tools::Utils::ToDegrees(euler.x);
581  euler.y = Tools::Utils::ToDegrees(euler.y);
582  euler.z = Tools::Utils::ToDegrees(euler.z);
583 
584  return euler;
585  }

592  inline Quaternion Quaternion::ToQuaternion(const double p_yaw, const double p_pitch, const double p_roll)
593  {
594  Quaternion result;
595 
596  const double cosYaw = Tools::Utils::Cos(p_yaw * 0.5);
597  const double sinYaw = Tools::Utils::Sin(p_yaw * 0.5);
598  const double cosPitch = Tools::Utils::Cos(p_pitch * 0.5);
599  const double sinPitch = Tools::Utils::Sin(p_pitch * 0.5);
600  const double cosRoll = Tools::Utils::Cos(p_roll * 0.5);
601  const double sinRoll = Tools::Utils::Sin(p_roll * 0.5);
602 
603  result.w = cosYaw * cosPitch * cosRoll + sinYaw * sinPitch * sinRoll;
604  result.axis.x = cosYaw * cosPitch * sinRoll - sinYaw * sinPitch * cosRoll;
605  result.axis.y = sinYaw * cosPitch * sinRoll + cosYaw * sinPitch * cosRoll;
606  result.axis.z = sinYaw * cosPitch * cosRoll - cosYaw * sinPitch * sinRoll;
607 
608  return { result };
609  }

622  inline Matrix3<float> Quaternion::ToMatrix3() const
623  {
624  Matrix3<float> result;
625 
626  const float fw = static_cast<float>(w);
627  Vector3<float> faxis{};
628  faxis.x = static_cast<float>(axis.x);
629  faxis.y = static_cast<float>(axis.y);
630  faxis.z = static_cast<float>(axis.z);
631 
632  result.m_data[0] = 2.0f * (fw * fw + faxis.x * faxis.x) - 1.0f;
633  result.m_data[3] = 2.0f * (faxis.x * faxis.y - fw * faxis.z);
634  result.m_data[6] = 2.0f * (faxis.x * faxis.z + fw * faxis.y);
635 
636  result.m_data[1] = 2.0f * (faxis.x * faxis.y + fw * faxis.z);
637  result.m_data[4] = 2.0f * (fw * fw + faxis.y * faxis.y) - 1.0f;
638  result.m_data[7] = 2.0f * (faxis.y * faxis.z - fw * faxis.x);
639 
640  result.m_data[2] = 2.0f * (faxis.x * faxis.z - fw * faxis.y);
641  result.m_data[5] = 2.0f * (faxis.y * faxis.z + fw * faxis.x);
642  result.m_data[8] = 2.0f * (fw * fw + faxis.z * faxis.z) - 1.0f;
643 
644  return result;
645  }

647  inline Matrix4<float> Quaternion::ToMatrix4() const
648  {
649  Matrix4<float> result{};
650  const float sqw = static_cast<float>(w * w);
651  const float sqx = static_cast<float>(axis.x * axis.x);
652  const float sqy = static_cast<float>(axis.y * axis.y);
653  const float sqz = static_cast<float>(axis.z * axis.z);
654 
655  // invs (inverse square length) is only required if quaternion is not already normalised
656  const float invs = 1.0f / (sqx + sqy + sqz + sqw);
657  result.m_data[0] = (sqx - sqy - sqz + sqw) * invs; // since sqw + sqx + sqy + sqz =1/invs*invs
658  result.m_data[5] = (-sqx + sqy - sqz + sqw) * invs;
659  result.m_data[10] = (-sqx - sqy + sqz + sqw) * invs;
660 
661  float tmp1 = static_cast<float>(axis.x * axis.y);
662  float tmp2 = static_cast<float>(axis.z * w);
663  result.m_data[4] = 2.0f * (tmp1 + tmp2) * invs;
664  result.m_data[1] = 2.0f * (tmp1 - tmp2) * invs;
665 
666  tmp1 = static_cast<float>(axis.x * axis.z);
667  tmp2 = static_cast<float>(axis.y * w);
668  result.m_data[8] = 2.0f * (tmp1 - tmp2) * invs;
669  result.m_data[2] = 2.0f * (tmp1 + tmp2) * invs;
670  tmp1 = static_cast<float>(axis.y * axis.z);
671  tmp2 = static_cast<float>(axis.x * w);
672  result.m_data[9] = 2.0f * (tmp1 + tmp2) * invs;
673  result.m_data[6] = 2.0f * (tmp1 - tmp2) * invs;
674 
675  return { result };
676  }
